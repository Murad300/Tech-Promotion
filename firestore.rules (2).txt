rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Global Safety Net
    match /{document=**} {
      allow read, write: if false;
    }

    function isValidId(id) {
      return id is string && id.size() <= 128 && id.matches('^[a-zA-Z0-9_\\-]+$');
    }

    // Schema validation helper for a Document
    function isValidDocument(data) {
      return data.keys().hasAll(['title', 'content', 'project', 'category', 'status', 'author', 'createdAt', 'updatedAt'])
        && data.keys().size() == 8
        && data.title is string && data.title.size() <= 100
        && data.project is string && data.project.size() <= 50
        && data.category is string && data.category in ['Specs', 'Meeting Notes', 'Design', 'Report', 'Proposal', 'General']
        && data.status is string && data.status in ['Draft', 'In Review', 'Approved', 'Archived']
        && data.content is string && data.content.size() <= 50000
        && data.author is string && data.author.size() <= 50;
    }

    // Schema validation for a Contact
    function isValidContact(data) {
      return data.keys().hasAll(['phoneNumber', 'email', 'createdAt'])
        && data.phoneNumber is string && data.phoneNumber.size() <= 20
        && data.email is string && data.email.size() <= 100;
    }

    match /documents/{documentId} {
      allow read: if true;
      allow create: if isValidId(documentId) && isValidDocument(request.resource.data);
      allow update: if isValidId(documentId) && isValidDocument(request.resource.data) && request.resource.data.createdAt == resource.data.createdAt;
      allow delete: if true;
    }

    match /contacts/{contactId} {
      allow read, create, update, delete: if true;
    }

    match /promo_templates/{templateId} {
      allow read, create, update, delete: if true;
    }
  }
}
